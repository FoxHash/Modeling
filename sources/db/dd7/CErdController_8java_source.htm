281  @RequestMapping( value = "/normalization/{model}", produces = MediaType.APPLICATION_JSON_VALUE )
282  public Object normaizationCheck( @PathVariable( "model" ) final String p_model )
283  {
284  final Map<String, IEntity<IAttribute>> l_nonnormalizedentities = EModelStorage.INSTANCE.apply( p_model ).<IErd>raw().checkNormalization();
285  final Map<Object, Object> l_responsemap = new HashMap<>();
286  final DateFormat l_dateformat = new SimpleDateFormat( "dd.MM.yyyy HH:mm:ss" );
287  final Date l_date = new Date();
288 
289  if ( l_nonnormalizedentities.size() > 0 )
290  {
291  l_responsemap.put( "description", "the following entities can't be normalized, because they have no key attribute connected" );
292  l_responsemap.put( "date/time", l_dateformat.format( l_date ) );
293  l_responsemap.put( "error counter", l_nonnormalizedentities.size() );
294  l_responsemap.put( "non normalized entities", l_nonnormalizedentities );
295 
296  return new ResponseEntity<>( l_responsemap, HttpStatus.BAD_REQUEST );
297  }
298  else
299  {
300  l_responsemap.put( "date/time", l_dateformat.format( l_date ) );
301  l_responsemap.put( "description", "all entities contains at least one key attribute" );
302 
303  return new ResponseEntity<>( l_responsemap, HttpStatus.OK );
304  }
