174  return this;

291  CSerializer( @NonNull final IGraph<IErdNode, IErdEdge> p_model )
292  {
293  m_entities = p_model.nodes()
294  .filter( i -> i instanceof IEntity )
295  .map( INode::<IEntity>raw )
296  .collect( Collectors.toMap( INode::id, i -> i ) );
297 
298  m_relationships = p_model.nodes()
299  .filter( i -> i instanceof IRelationship )
300  .map( INode::<IRelationship>raw )
301  .collect( Collectors.toMap( INode::id, i -> i ) );
302 
303  m_inheritrelationships = p_model.nodes()
304  .filter( i -> i instanceof IInheritRelationship )
305  .map( INode::<IInheritRelationship>raw )
306  .collect( Collectors.toMap( INode::id, i -> i ) );
307 
308  m_stats.put( "entities", m_entities.size() );
309  m_stats.put( "relationships", m_relationships.size() );
310  m_stats.put( "inherit_relationships", m_inheritrelationships.size() );
311  }
