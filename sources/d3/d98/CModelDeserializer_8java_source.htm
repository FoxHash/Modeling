88  private static void deserializeEntities( @NonNull final String p_modelid, final JsonNode p_jsonnode )
89  {
90 
91  p_jsonnode.get( "entities" ).elements().forEachRemaining( entity ->
92  {
93  final String l_id = ( entity.get( ID ).isNull() ) ? null : entity.get( ID ).asText();
94  final boolean l_weakentity = entity.get( "weak_entity" ).asBoolean();
95 
96  if ( !entity.has( "attributes" ) )
97  {
98  throw new RuntimeException( MessageFormat.format( "the entity [{0}] has no attributes connected", l_id ) );
99  }
100 
101  if ( !entity.get( "attributes" ).elements().hasNext() )
102  {
103  throw new RuntimeException( MessageFormat.format( "the entity [{0}] has no attributes connected", l_id ) );
104  }
105 
106  EModelStorage.INSTANCE.apply( p_modelid ).<IErd>raw().addEntity( l_id, l_weakentity );
107 
108  if ( entity.has( "attributes" ) && entity.get( "attributes" ).elements().hasNext() )
109  {
110  entity.get( "attributes" ).elements().forEachRemaining( attribute ->
111  {
112 
113  final String l_name = ( attribute.get( "name" ).isNull() ) ? null : attribute.get( "name" ).asText();
114  final String l_property = attribute.get( "property" ).isNull() ? null : attribute.get( "property" ).asText();
115 
116  EModelStorage.INSTANCE.apply( p_modelid ).<IErd>raw().addAttributeToEntity( l_name, l_property, l_id );
117 
118  } );
119  }
120  } );
121 
122  }
