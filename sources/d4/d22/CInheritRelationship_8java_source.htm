48  private final List<IEntity<IAttribute>> m_childentities = new ArrayList<>();

55  @Override
56  public IEntity<IAttribute> setParentEntity( @NonNull final IEntity<IAttribute> p_entity )
57  {
58  if ( m_parententity.compareAndSet( null, p_entity ) )
59  {
60  return p_entity;
61  }
62  else
63  {
64  throw new RuntimeException( "You set already a parent entity for this relationship" );
65  }
66  }
67 
68  @Override
69  public String getParentEntity()
70  {
71  if ( Objects.isNull( m_parententity.get() ) )
72  {
73  return "";
74  }
75  return m_parententity.get().id();
76  }
77 
78  @Override
79  public List<String> getChildEntities()
80  {
81  final List<String> l_childnames = new ArrayList<>();
82  m_childentities.forEach( elm -> l_childnames.add( elm.id() ) );
83  return Collections.unmodifiableList( l_childnames );
84  }
85 
86  @Override
87  public IEntity<IAttribute> connectChildEntity( @NonNull final IEntity<IAttribute> p_entity )
88  {
89  if ( m_childentities.contains( p_entity ) )
90  {
91  throw new RuntimeException( MessageFormat.format( "You already set this entity[{0}] to the child entities", p_entity.id() ) );
92  }
93  m_childentities.add( p_entity );
94  return p_entity;
95  }
96 
97  @Override
98  public boolean isValidISARelationship()
99  {
100  return ( Objects.nonNull( m_parententity.get() ) ) && m_childentities.size() >= 1;
101  }
