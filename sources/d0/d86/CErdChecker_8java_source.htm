68  @Override
69  public boolean validateRelationships()
70  {
71  final long l_allrelationships = m_model.nodes()
72  .filter( i -> i instanceof IRelationship )
73  .count();
74 
75  // relationship is valid if the two entities are set
76  final long l_validrelationship = m_model.nodes()
77  .filter( i -> i instanceof IRelationship )
78  .filter(
79  i -> i.<IRelationship<IAttribute>>raw().getConnectedEntities().size() == 2
80  || i.<IRelationship<IAttribute>>raw().isRecursive() )
81  .count();
82 
83  m_model.nodes()
84  .filter( i -> i instanceof IRelationship )
85  .filter( i -> i.<IRelationship<IAttribute>>raw().getConnectedEntities().size() <= 1 && !i.<IRelationship<IAttribute>>raw().isRecursive() )
86  .forEach( i -> m_errors.add( i.id() + ERROR_RELATIONSHIP ) );
87 
88  return l_allrelationships == l_validrelationship;
89  }
90 
91  @Override
92  public boolean validateEntities()
93  {
94  final long l_allentities = m_model.nodes()
95  .filter( i -> i instanceof IEntity )
96  .count();
97 
98  // entity is valid if it has more than one attribute
99  final long l_validentities = m_model.nodes()
100  .filter( i -> i instanceof IEntity )
101  .filter( i -> i.<IEntity<IAttribute>>raw().getConnectedAttributes().size() >= 1 )
102  .count();
103 
104  m_model.nodes()
105  .filter( i -> i instanceof IEntity )
106  .filter( i -> i.<IEntity<IAttribute>>raw().getConnectedAttributes().size() < 1 )
107  .forEach( i -> m_errors.add( i.id() + ERROR_ENTITYATTRIBUT ) );
108 
109  return l_allentities == l_validentities;
110  }
111 
112  @Override
113  public boolean validateISARelationships()
114  {
115  final List<String> l_childnames = new ArrayList<>();
116  final List<String> l_relationshipconnections = new ArrayList<>();
117  final List<String> l_invalidchilds;
118 
119  m_model.nodes()
120  .filter( i -> i instanceof IRelationship )
121  .forEach( i -> l_relationshipconnections.addAll( i.<IRelationship<IAttribute>>raw().getConnectedEntities().keySet() ) );
122 
123 
124  m_model.nodes()
125  .filter( i -> i instanceof IInheritRelationship )
126  .forEach( i -> l_childnames.addAll( i.<IInheritRelationship<IAttribute>>raw().getChildEntities() ) );
127 
128 
129  final long l_invalidisarelationships = l_childnames.stream().filter( l_relationshipconnections::contains ).count();
130 
131  l_invalidchilds = l_childnames.stream()
132  .filter( l_relationshipconnections::contains )
133  .collect( Collectors.toList() );
134 
135  l_invalidchilds.forEach( i -> m_errors.add( i + ERROR_ISARELATIONSHIP ) );
136 
137  return l_invalidisarelationships == 0;
138  }
139 
140  @Override
141  public boolean validateWeakEntityKeys()
142  {
143  final AtomicReference<Integer> l_counter = new AtomicReference<>( 0 );
144 
145  m_model.nodes()
146  .filter( i -> i instanceof IEntity )
147  .filter( i -> i.<IEntity<IAttribute>>raw().isWeakEntity() )
148  .forEach( p_iErdNode -> p_iErdNode.<IEntity<IAttribute>>raw().getConnectedAttributes().forEach( ( p_str, p_attribute ) ->
149  {
150  if ( p_attribute.getProperty().equals( EAttributeProperty.KEY.getProperty() ) )
151  {
152  l_counter.set( l_counter.get() + 1 );
153  m_errors.add( p_attribute.attributeName() + " (" + p_iErdNode.id() + ")" + ERROR_WEAKATTRIBUTEENTITY );
154  }
155  } ) );
156 
157  return l_counter.get() == 0;
158  }
159 
160  @Override
161  @SuppressWarnings( "unchecked" )
162  public Map<String, IEntity<IAttribute>> validateNormalization()
163  {
164  final Map<String, IEntity<IAttribute>> l_nonnormalizedentities;
165 
166  l_nonnormalizedentities = m_model.nodes()
167  .filter( i -> i instanceof IEntity )
168  .map( INode::<IEntity>raw )
169  .collect( Collectors.toMap( INode::id, i -> i ) );
170 
171  m_model.nodes()
172  .filter( i -> i instanceof IEntity )
173  .forEach( p_iErdNode -> p_iErdNode.<IEntity<IAttribute>>raw().getConnectedAttributes().forEach( ( p_str, p_iattribute ) ->
174  {
175  if ( p_iattribute.getProperty().equals( EAttributeProperty.KEY.getProperty() ) )
176  {
177  l_nonnormalizedentities.remove( p_iErdNode.id() );
178  }
179  } ) );
180 
181  return Collections.unmodifiableMap( l_nonnormalizedentities );
182  }
